enum RenderType {
	RENDER_STATIC,
	RENDER_TERRAIN,
};

struct Renderable {
	GLuint VAO;
	GLuint program;
	GLsizei num_indices;
	
	enum RenderType type;
	union {
		struct StaticShader {
			GLuint transformation_matrix;
			GLuint projection_matrix;
			GLuint view_matrix;
			GLuint light_position;
			...
		} static_shader;
		
		struct TerrainShader {
			GLuint transformation_matrix;
			GLuint projection_matrix;
			GLuint view_matrix;
			GLuint light_position;
			...
		} terrain_shader;
	};
};
